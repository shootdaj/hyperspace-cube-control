    ```typescript
    export class WLEDPaintOutput {
      private lastSent = new Uint8Array(480 * 3);
      private pendingBuffer: Uint8Array | null = null;
      private throttleTimer: ReturnType<typeof setTimeout> | null = null;
      private readonly THROTTLE_MS = 33; // ~30fps

      sendPaint(buffer: Uint8Array): void {
        // Store pending; if no throttle active, send immediately
        this.pendingBuffer = buffer;
        if (!this.throttleTimer) {
          this.flush();
          this.throttleTimer = setTimeout(() => {
            this.throttleTimer = null;
            if (this.pendingBuffer) this.flush();
          }, this.THROTTLE_MS);
        }
      }

      private flush(): void {
        if (!this.pendingBuffer) return;
        const buf = this.pendingBuffer;
        this.pendingBuffer = null;
        
        // Find changed LED indices
        const changed: number[] = [];
        for (let i = 0; i < 480; i++) {
          const off = i * 3;
          if (buf[off] !== this.lastSent[off] ||
              buf[off+1] !== this.lastSent[off+1] ||
              buf[off+2] !== this.lastSent[off+2]) {
            changed.push(i);
          }
        }
        if (changed.length === 0) return;

        // Build contiguous ranges
        const ranges = buildRanges(changed);
        // Send each range as seg.i payload
        for (const range of ranges) {
          const payload: number[] = [];
          for (let i = range.start; i <= range.end; i++) {
            payload.push(buf[i*3], buf[i*3+1], buf[i*3+2]);
          }
          WLEDWebSocketService.getInstance().send({
            seg: [{ id: 0, i: [range.start, ...payload] }],
          });
        }
        
        // Update lastSent
        this.lastSent.set(buf);
      }

      sendAll(buffer: Uint8Array): void { /* full send in chunks */ }
      reset(): void { this.lastSent.fill(0); }
      destroy(): void { if (this.throttleTimer) clearTimeout(this.throttleTimer); }
    }
    ```

    Helper `buildRanges(sorted: number[]): {start: number, end: number}[]` groups contiguous indices.
