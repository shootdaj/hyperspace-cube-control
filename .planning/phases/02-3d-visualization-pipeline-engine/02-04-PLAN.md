<OrbitControls
  enableDamping          // Smooth deceleration after drag
  dampingFactor={0.05}   // Controls how quickly damping decays (0.05 = smooth)
  minDistance={0.8}      // Closest zoom — don't go inside the cube

    TWO: THREE.TOUCH.DOLLY_PAN,   // Two fingers = pinch zoom (DOLLY_PAN = zoom only since pan disabled)
  }}
  makeDefault            // Registers as default camera controls for the Canvas
/>
